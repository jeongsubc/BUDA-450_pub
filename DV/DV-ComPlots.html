11  Common Plots

This section introduces to various types of plots using popular visualization packages:

  • Matplotlib (https://matplotlib.org/): This is the grandparent of Python visualization libraries. It’s incredibly powerful and provides a high degree of control over every aspect of a plot, making it suitable for creating publication-quality figures. While it can be verbose for simple plots, many other libraries are built on top of Matplotlib, leveraging its capabilities.

  • Seaborn (https://seaborn.pydata.org/): Built on Matplotlib, Seaborn provides a higher-level interface for drawing attractive and informative statistical graphics. It simplifies the creation of complex plots like heatmaps, violin plots, pair plots, and categorical plots, making it a favorite for data exploration and analysis, especially when working with pandas.DataFrames.

  • Altair (https://altair-viz.github.io/): Altair is a declarative statistical visualization library based on Vega-Lite. It focuses on providing a simple API where users define the structure of the visualization through data and chart specifications. Altair’s concise syntax makes it ideal for quickly prototyping and exploring data interactively, especially with smaller to medium-sized datasets.

  • Plotly (https://plotly.com/python/): Plotly is a versatile, open-source graphing library that enables the creation of interactive, web-based visualizations. It supports a wide range of chart types (2D, 3D, statistical, scientific, financial, geographic) and integrates seamlessly with Jupyter notebooks, Dash applications, and other web frameworks. Plotly Express, a high-level API, makes it very easy to generate common interactive plots with minimal code.

# custom function...
def radar_plot(dataframe, figsize=(12,8)):
    rows, cols = dataframe.index, dataframe.columns
    n_cols = len(cols)

    # Angles for each column (variable)
    angles = [n / float(n_cols) * 2 * np.pi for n in range(n_cols)]
    angles += [angles[0]]

    # Initialise spider plot
    fig = plt.figure(figsize=figsize)
    ax = fig.add_subplot(111, polar=True)

    # Plot each row (observation) of the DataFrame
    for idx, row in dataframe.iterrows():
        values = row.tolist()
        values += [values[0]]  # close loop

        ax.plot(angles, values, linewidth=1, linestyle="solid", label=idx)
        ax.fill(angles, values, alpha=0.1)  # optional fill

    # Add category labels
    ax.set_xticks(angles[:-1])
    ax.set_xticklabels(cols)

    ax.legend(loc="upper right", bbox_to_anchor=(1.1, 1.1))
    plt.show()

def radar_plot_seaborn(df, figsize=(12, 8), fill=True):
    cols = df.columns.tolist()
    n_cols = len(cols)
    angles = [n / float(n_cols) * 2 * np.pi for n in range(n_cols)]
    angles += [angles[0]]

    sns.set_theme(style="whitegrid")
    fig, ax = plt.subplots(subplot_kw={"projection": "polar"}, figsize=figsize)

    for idx, row in df.iterrows():
        values = row.tolist() + [row.tolist()[0]]
        ax.plot(angles, values, linewidth=2, label=idx)
        if fill:
            ax.fill(angles, values, alpha=0.15)

    ax.set_xticks(angles[:-1])
    ax.set_xticklabels(cols)
    ax.set_title("Radar Chart (Seaborn + Matplotlib)", pad=20)
    ax.legend(loc="upper right", bbox_to_anchor=(1.15, 1.15))
    plt.show()

def radar_plot_altair(df, title="Radar Chart (Altair via Cartesian)"):
    cols = df.columns.tolist()
    n_cols = len(cols)
    angles = np.linspace(0, 2*np.pi, n_cols, endpoint=False)

    # Build long-form with Cartesian coords per row/variable and close the loop
    records = []
    for row_name, row in df.iterrows():
        vals = row.values
        xs = (vals * np.cos(angles)).tolist()
        ys = (vals * np.sin(angles)).tolist()
        # close loop
        xs.append(xs[0]); ys.append(ys[0])
        vars_seq = cols + [cols[0]]
        for order, (x, y, var, val) in enumerate(zip(xs, ys, vars_seq, list(vals)+[vals[0]])):
            records.append({
                "Row": row_name, "Variable": var, "Value": float(val),
                "x": float(x), "y": float(y), "order": order
            })
    data = pd.DataFrame(records)

    line = alt.Chart(data).mark_line().encode(
        x="x:Q", y="y:Q", color="Row:N", detail="Row:N", order="order:O"
    )
    pts = alt.Chart(data).mark_point().encode(
        x="x:Q", y="y:Q", color="Row:N"
    )
    chart = (line + pts).properties(
        width=450, height=450, title=title
    )
    chart

def radar_plot_plotly(df, title="Radar Chart (Plotly)"):
    categories = df.columns.tolist()
    fig = go.Figure()
    for idx, row in df.iterrows():
        values = row.tolist() + [row.tolist()[0]]              # close loop
        fig.add_trace(go.Scatterpolar(
            r=values,
            theta=categories + [categories[0]],
            mode="lines+markers",
            name=idx
        ))
    fig.update_layout(
        title=title,
        polar=dict(radialaxis=dict(visible=True)),
        showlegend=True
    )
    fig.show()
